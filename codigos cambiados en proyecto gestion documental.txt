@app.route(
    '/proyecto/contrato',
    methods=['GET', 'POST']
)
def contrato():

    proyecto = Proyecto.query.get_or_404(id)

    cliente = Cliente.query.get_or_404(
        proyecto.cliente_id
    )

    meses = (
        (proyecto.fecha_fin.year - proyecto.fecha_inicio.year) * 12
        + (proyecto.fecha_fin.month - proyecto.fecha_inicio.month)
    )

    if proyecto.fecha_fin.day >= proyecto.fecha_inicio.day:
        meses += 1

    if request.method == 'POST':

        contexto = {

            "CLIENTE": cliente.nombre_cliente,
            "RFC": cliente.rfc,
            "REPRESENTANTE": cliente.representante,

            "PROYECTO": proyecto.nombre_proyecto,
            "TIPO_SERVICIO": proyecto.tipo_servicio,
            "MONTO": proyecto.monto,

            "INICIO":
                proyecto.fecha_inicio.strftime("%d/%m/%Y")
                if proyecto.fecha_inicio else "",

            "FIN":
                proyecto.fecha_fin.strftime("%d/%m/%Y")
                if proyecto.fecha_fin else "",

            "OBJETO":
                request.form['objeto'],

            "FORMA_PAGO":
                request.form['forma_pago'],

            "VIGENCIA":
                f"{meses} Meses",

            "LUGAR_FIRMA": request.form['lugar_firma']
        }

        archivo = generar_contrato(contexto)

        return send_file(
            archivo,
            as_attachment=True
        )

    return render_template(
        'contrato_form.html',
        proyecto=proyecto,
        cliente=cliente
    )
